We now create the function

auto_t_test <- function(data, var_name, factor, order_vec){
  df_out <- data |> 
    t_test(formula = paste(var_name, " ~ ", factor) |> as.formula(),
          order = order_vec,
         alternative = "two-sided") |>
  mutate(variable = var_name)
  return(df_out)
}
